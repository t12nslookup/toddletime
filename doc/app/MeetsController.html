def edit
  if @meet.rota_leaders.count.zero? && @meet.meet_type.present?

def index
  @meets = Meet
  if params[:historic] == 'true'
    @meets = @meets.in_reverse.find_historic
  else
    @meets = @meets.in_order.find_future
  end
end
